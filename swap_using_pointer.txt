#include<stdio.h>

void swap_mine(int *ip, int *jp);
main()
{
    int i=5;
    int j=6;
    printf("before swaping i = %d , j = %d \n",i,j);
    swap_mine(&i,&j);
    printf("after  swaping i = %d , j = %d",i,j);
}

void swap_mine(int *ip, int *jp)
{
    int temp;
    temp = *ip;
    *ip  = *jp;
    *jp  = temp;
}
