#include <stdio.h>

typedef struct
{
    char fn[20];
    char ln[20];
    int  ec;
}emp;

void print_emp(emp e_local);

void main(void)
{
    emp elist[3]={
                    {"Arindam","Sarkar",1},
                    {"Ram","Kumar",2},
                    {"Sam","Last",3}
                 };
    int i=0;

    for(i=0;i<3;i++)
    {
        print_emp(elist[i]);
    }
}

void print_emp(emp e_local)
{
    printf("first name = %s \n",e_local.fn);
    printf("last  name = %s \n",e_local.ln);
    printf("Emp   Code = %d \n",e_local.ec);
    printf("\n");
}
