#include <stdio.h>

struct employee
{
    char fn[20];
    char ln[20];
    int  ec;
};

void main(void)
{
    struct employee elist[2]={
                                {"Arindam","Sarkar",1},
                                {"Ram","Kumar",2}
                             };
    int i=0;

    for(i=0;i<2;i++)
    {
        printf("first name = %s \n",elist[i].fn);
        printf("last  name = %s \n",elist[i].ln);
        printf("Emp   Code = %d \n",elist[i].ec);
        printf("\n\n");
    }

}
